  <script>
  
    const dataInicio = new Date(2023, 4, 15, 0, 0, 0); 

    const contador = document.getElementById("contador");

    function plural(n, palavra) {
      return n === 1 ? palavra : palavra + "s";
    }

    function diferencaDetalhada(inicio, fim) {
      
      let anos = fim.getFullYear() - inicio.getFullYear();
      let meses = fim.getMonth() - inicio.getMonth();
      let dias = fim.getDate() - inicio.getDate();

      if (dias < 0) {
        
        const ultimoMes = new Date(fim.getFullYear(), fim.getMonth(), 0);
        dias += ultimoMes.getDate();
        meses -= 1;
      }
      if (meses < 0) {
        meses += 12;
        anos -= 1;
      }

      return { anos, meses, dias };
    }

    function atualizarContador() {
      const agora = new Date();
      const diffMs = agora - dataInicio;

      if (diffMs < 0) {
        contador.textContent = "Ainda não começamos! ??";
        return;
      }

    
      const totalSeg = Math.floor(diffMs / 1000);
      const totalMin = Math.floor(totalSeg / 60);
      const totalHoras = Math.floor(totalMin / 60);
      const totalDias = Math.floor(totalHoras / 24);

      const horas = totalHoras % 24;
      const minutos = totalMin % 60;
      const segundos = totalSeg % 60;

      const det = diferencaDetalhada(dataInicio, agora);

      contador.innerHTML = `
        ${det.anos} ${plural(det.anos, "ano")}, 
        ${det.meses} ${plural(det.meses, "mês")}, 
        ${det.dias} ${plural(det.dias, "dia")}<br>
        (${totalDias} ${plural(totalDias, "dia")}, ${horas} ${plural(horas, "hora")}, ${minutos} ${plural(minutos, "minuto")} e ${segundos} ${plural(segundos, "segundo")})
      `;
    }

    atualizarContador();
    setInterval(atualizarContador, 1000);
  </script>
